<script>

function sendMessage() {
    let inputField = document.getElementById("userInput");
    let userText = inputField.value.trim();

    if (userText === "") {
        alert("Please enter a query.");
        return;
    }

    displayMessage(userText, "user");

    let botResponse = getBotResponse(userText);

    setTimeout(() => {
        displayMessage(botResponse, "bot");
        logToDatabase(userText, botResponse);
    }, 500);

    inputField.value = "";
}

// Display messages
function displayMessage(text, sender) {
    let chatBox = document.getElementById("chatBox");

    let msg = document.createElement("div");
    msg.classList.add("message", sender);
    msg.innerText = text;

    chatBox.appendChild(msg);
    chatBox.scrollTop = chatBox.scrollHeight;
}

// 5.7.3 + 5.7.4 Response + Error Handling
function getBotResponse(input) {
    input = input.toLowerCase();

    if (input.includes("courses")) {
        return "We offer courses in Data Science, AI, and Web Development.";
    }
    else if (input.includes("enroll")) {
        return "You can enroll through our website by clicking on the 'Enroll Now' button.";
    }
    else if (input.includes("payment")) {
        return "We support online payments via card, UPI, and net banking.";
    }
    else {
        return "Sorry, I didn’t understand your query.";
    }
}

// 5.7.5 Database Simulation
function logToDatabase(query, response) {
    let db = document.getElementById("dbLog");

    let logEntry = document.createElement("div");
    logEntry.innerHTML = "<b>Q:</b> " + query + "<br><b>A:</b> " + response + "<hr>";

    db.appendChild(logEntry);
    db.scrollTop = db.scrollHeight;
}
